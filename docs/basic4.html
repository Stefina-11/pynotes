import string

def is_pangram(s):
    s = s.lower()
    return all(ch in s for ch in string.ascii_lowercase)

print("Pangram" if is_pangram(input("Enter sentence: ")) else "Not Pangram")

Enter sentence:  The quick brown fox jumps over the lazy dog


Pangram
