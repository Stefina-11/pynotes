class Counter:
    def __init__(self, limit):
        self.limit = limit
        self.num = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.num < self.limit:
            self.num += 1
            return self.num
        else:
            raise StopIteration

for i in Counter(3):
    print(i)

class Singleton:

class Publisher:
    def __init__(self):
        self.subscribers = []

    def subscribe(self, sub):
        self.subscribers.append(sub)

    def notify(self):
        for sub in self.subscribers:
            sub.update()

class Subscriber:
    def update(self):
        print("Subscriber updated!")

p = Publisher()
s = Subscriber()

class BankAccount:
    def __init__(self, owner, balance=0):
